# Database Schema

## users
- id: UUID PRIMARY KEY (auto-generated by database)
- officer_id: VARCHAR(6) UNIQUE NOT NULL (6-digit ID for login, e.g., "001234")
- email: VARCHAR(255) UNIQUE NOT NULL
- password_hash: VARCHAR(255) NOT NULL
- role: VARCHAR(20) NOT NULL CHECK (role IN ('officer', 'admin'))
- full_name: VARCHAR(255)
- created_at: TIMESTAMP DEFAULT NOW()
- updated_at: TIMESTAMP DEFAULT NOW()

## drivers
- id: UUID PRIMARY KEY
- full_name: VARCHAR(255) NOT NULL
- license_no: VARCHAR(50) UNIQUE NOT NULL
- plate_no: VARCHAR(20)
- contact: VARCHAR(100)
- status: VARCHAR(20) DEFAULT 'Active' CHECK (status IN ('Active', 'Flagged', 'Suspended'))
- face_embedding: JSONB (stores array of floats from facial recognition)
- strike_count: INTEGER DEFAULT 0
- created_at: TIMESTAMP DEFAULT NOW()
- updated_at: TIMESTAMP DEFAULT NOW()

## offences
- id: UUID PRIMARY KEY
- driver_id: UUID REFERENCES drivers(id) NOT NULL
- officer_id: UUID REFERENCES users(id) NOT NULL
- offence_type_id: UUID REFERENCES offence_types(id) NOT NULL
- fine_amount: DECIMAL(10,2) NOT NULL
- strike_delta: INTEGER NOT NULL (how many strikes this offence added)
- issued_at: TIMESTAMP DEFAULT NOW()
- notes: TEXT

## offence_types
- id: UUID PRIMARY KEY
- name: VARCHAR(100) NOT NULL
- description: TEXT
- base_fine: DECIMAL(10,2) NOT NULL
- strike_weight: INTEGER NOT NULL
- severity: VARCHAR(20) NOT NULL
- is_active: BOOLEAN DEFAULT TRUE
- created_at: TIMESTAMP DEFAULT NOW()

## penalty_rules
- id: UUID PRIMARY KEY
- min_strikes: INTEGER NOT NULL
- max_strikes: INTEGER NOT NULL
- fine_multiplier: DECIMAL(3,2) NOT NULL (e.g., 1.5 for 50% increase)
- status_flag: VARCHAR(50) (e.g., 'Warning', 'Flagged', 'Suspended')
- created_at: TIMESTAMP DEFAULT NOW()

## audit_logs
- id: UUID PRIMARY KEY
- user_id: UUID REFERENCES users(id)
- action: VARCHAR(100) NOT NULL (e.g., 'DRIVER_CREATED', 'OFFENCE_ISSUED')
- entity_type: VARCHAR(50) NOT NULL (e.g., 'driver', 'offence')
- entity_id: UUID
- timestamp: TIMESTAMP DEFAULT NOW()
- metadata: JSONB (stores additional context about the action)
